Temperature Conversion

This Ruby program will prompt the user for a temperature in degrees Fahrenheit and let the user know what the corresponding temperature is in Celsius. 

def f_to_c(fahren)
  fahren = fahren.to_f
  fahren = fahren - 32
  celsius = fahren * 5/9
  return celsius.round(2)
end

print "Enter a temperature in Fahrenheit: "
temp = gets.chomp

puts "In Celsius, that would be approximately #{f_to_c(temp)}."

def ordinal(right, number)
    og_number = number

    if number > 99
        number = number.to_s
        number = number[-2]
        number = number.to_i
    else
        number = number.to_s
        number = number[0]
        number = number.to_i
    end

    if number == 1 && og_number > 1
        return og_number.to_s + "th"
    end

    if right == 1
        return og_number.to_s + "st"
    elsif right == 2
        return og_number.to_s + "nd"
    elsif right == 3
        return og_number.to_s + "rd"
    else
     return og_number.to_s + "th"
    end
end

puts "That's the #{ordinal(right_digit, input)} number!"

Foobar

This Ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

print "How many items do you want to see?: "
items = gets.chomp.to_i
x = 0
numbers = []

while x < items do
    x = x + 1
    numbers << x
end


def foobar(number)
    output = []
    number.each do |digit|
        if digit % 3 == 0
            if digit % 5 == 0
                output << "Foobar!"
            else
                output << "Foo"
            end
        elsif digit % 5 == 0
                output << "Bar"
        else
            output << digit
        end
    end
    print output
    puts ""
end

foobar(numbers)
